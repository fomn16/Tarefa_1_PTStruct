create_table :users do |t|
      t.string :name
      t.string :nickname
      t.string :email
      t.date :birthday
      t.references :gender, foreign_key: true
      t.int :level
      t.references :group, foreign_key: true
      t.string :birthLocation
      t.integer :coins

      t.timestamps

create_table :sub_groups do |t|
      t.string :name
      t.references :group, foreign_key: true

      t.timestamps
